PROBLEM
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  User creates game with 10 rounds → Game ends too early
  
ROOT CAUSE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  Off-by-one error: Confusion between 0-indexed array positions and 
  1-indexed human-readable question counts.
  
  ❌ OLD CODE:
    if (nextQuestionIndex >= gameRoom.maxQuestions) {
      // END GAME
    }
  
  When maxQuestions = 5 and currentIndex = 4:
    nextIndex = 5
    Check: 5 >= 5? YES → END (but should play round 5!)
  
  ✅ FIXED CODE:
    if (nextQuestionIndex > gameRoom.maxQuestions) {
      // END GAME
    }
  
  Now: 5 > 5? NO → CONTINUE (play round 5) ✓

SOLUTION
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  Changed >= to > in two locations in GameRound.tsx
  
  File: src/pages/GameRound.tsx
  
  Location 1 (line ~170):
    In host's auto-advance effect
    if (nextQuestionIndex > gameRoom.maxQuestions)
  
  Location 2 (line ~238):
    In all-players game-end check
    if (nextQuestionIndex > gameRoom.maxQuestions)
